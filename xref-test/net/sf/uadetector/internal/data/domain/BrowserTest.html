132 	@Test
133 	public void copyTo_withOperatingSystem() {
134 		final SortedSet<OperatingSystemPattern> patternSet = new TreeSet<OperatingSystemPattern>();
135 		final OperatingSystem os = new OperatingSystem(1, "n1", "f1", "iu1", patternSet, "p1", "pu1", "u1", "i1");
136 		final Browser b = new Blueprint().operatingSystem(os).build();
137 		final Builder builder = new UserAgent.Builder();
138 		b.copyTo(builder);
139 		assertThat(builder.getFamily()).isEqualTo(b.getFamily());
140 		assertThat(builder.getProducer()).isEqualTo(b.getProducer());
141 		assertThat(builder.getProducerUrl()).isEqualTo(b.getProducerUrl());
142 		assertThat(builder.getTypeName()).isEqualTo(b.getType().getName());
143 		assertThat(builder.getUrl()).isEqualTo(b.getUrl());
144 		assertThat(builder.getOperatingSystem()).isNotNull();
145 	}

356 	@Test
357 	public void testGetters() {
358 		final int id = 12354;
359 		final String icon = "bunt.png";
360 		final String infoUrl = "http://programming-motherfucker.com/";
361 		final String url = "http://user-agent-string.info/";
362 		final UserAgentFamily family = UserAgentFamily.FIREFOX;
363 		final String producerUrl = "https://github.com/before";
364 		final String producer = "Our Values";
365 		final BrowserType type = new BrowserType(1, "Browser");
366 		final SortedSet<OperatingSystemPattern> osPatternSet = new TreeSet<OperatingSystemPattern>();
367 		final OperatingSystem operatingSystem = new OperatingSystem(1, "n1", "f1", "iu1", osPatternSet, "p1", "pu1", "u1", "i1");
368 		final SortedSet<BrowserPattern> patterns = new TreeSet<BrowserPattern>();
369 		final Browser b = new Browser(id, family, family.getName(), patterns, type, operatingSystem, icon, infoUrl, producer, producerUrl,
370 				url);
371 		assertThat(b.getFamily()).isEqualTo(family);
372 		assertThat(b.getIcon()).isEqualTo("bunt.png");
373 		assertThat(b.getId()).isEqualTo(12354);
374 		assertThat(b.getInfoUrl()).isEqualTo("http://programming-motherfucker.com/");
375 		assertThat(b.getOperatingSystem()).isEqualTo(operatingSystem);
376 		assertThat(b.getPatterns()).isEqualTo(patterns);
377 		assertThat(b.getProducer()).isEqualTo("Our Values");
378 		assertThat(b.getProducerUrl()).isEqualTo("https://github.com/before");
379 		assertThat(b.getType()).isEqualTo(type);
380 		assertThat(b.getUrl()).isEqualTo("http://user-agent-string.info/");
381 	}

383 	@Test
384 	public void testToString() {
385 		// reduces only some noise in coverage report
386 		final int id = 12354;
387 		final String icon = "bunt.png";
388 		final String infoUrl = "http://programming-motherfucker.com/";
389 		final String url = "http://user-agent-string.info/";
390 		final UserAgentFamily family = UserAgentFamily.FIREFOX;
391 		final String producerUrl = "https://github.com/before";
392 		final String producer = "Our Values";
393 		final BrowserType type = new BrowserType(1, "Browser");
394 		final SortedSet<OperatingSystemPattern> osPatternSet = new TreeSet<OperatingSystemPattern>();
395 		final OperatingSystem os = new OperatingSystem(1, "n1", "f1", "iu1", osPatternSet, "p1", "pu1", "u1", "i1");
396 		final SortedSet<BrowserPattern> patterns = new TreeSet<BrowserPattern>();
397 		final Browser b = new Browser(id, family, family.getName(), patterns, type, os, icon, infoUrl, producer, producerUrl, url);
398 
399 		assertThat(b.toString())
400 				.isEqualTo(
401 						"Browser [id=12354, family=FIREFOX, familyName=Firefox, patterns=[], type=BrowserType [id=1, name=Browser], operatingSystem=OperatingSystem [id=1, name=n1, family=f1, infoUrl=iu1, patterns=[], producer=p1, producerUrl=pu1, url=u1, icon=i1], icon=bunt.png, infoUrl=http://programming-motherfucker.com/, producer=Our Values, producerUrl=https://github.com/before, url=http://user-agent-string.info/]");
402 	}
