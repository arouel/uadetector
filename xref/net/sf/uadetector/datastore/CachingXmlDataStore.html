36  /**
37   * Implementation of a {@link DataStore} which is able to recover <em>UAS data</em> in XML format from a cache file. If
38   * the cache file is empty, the data will be read from the given data URL.<br>
39   * <br>
40   * You can also update the data of the store at any time if you trigger {@link CachingXmlDataStore#refresh()}.
41   * 
42   * @author André Rouél
43   */
44  public final class CachingXmlDataStore extends AbstractRefreshableDataStore {
45  
46  	/**
47  	 * Internal data store which will be used to load previously saved <em>UAS data</em> from a cache file.
48  	 */
49  	private static class CacheFileDataStore extends AbstractDataStore {
50  		protected CacheFileDataStore(final Data data, final DataReader reader, final URL dataUrl, final Charset charset) {
51  			super(data, reader, dataUrl, dataUrl, charset);
52  		}
53  	}

97  	 * Constructs a new instance of {@code CachingXmlDataStore} with the given arguments. The given {@code cacheFile}
98  	 * can be empty or filled with previously cached data in XML format. The file must be writable otherwise an
99  	 * exception will be thrown.
100 	 * 
101 	 * @param fallback
102 	 *            <em>UAS data</em> as fallback in case the data on the specified resource can not be read correctly
103 	 * @return new instance of {@link CachingXmlDataStore}
104 	 * @throws net.sf.qualitycheck.exception.IllegalNullArgumentException
105 	 *             if one of the given arguments is {@code null}
106 	 * @throws net.sf.qualitycheck.exception.IllegalStateOfArgumentException
107 	 *             if the given cache file can not be read
108 	 * @throws net.sf.qualitycheck.exception.IllegalStateOfArgumentException
109 	 *             if no URL can be resolved to the given given file
110 	 */
111 	@Nonnull
112 	public static CachingXmlDataStore createCachingXmlDataStore(@Nonnull final DataStore fallback) {
113 		return createCachingXmlDataStore(findOrCreateCacheFile(), fallback);

131 	 *             if no URL can be resolved to the given given file

136 				fallback);

139 	/**
140 	 * Constructs a new instance of {@code CachingXmlDataStore} with the given arguments. The given {@code cacheFile}
141 	 * can be empty or filled with previously cached data in XML format. The file must be writable otherwise an
142 	 * exception will be thrown.
143 	 * 
144 	 * @param cacheFile
145 	 *            file with cached <em>UAS data</em> in XML format or empty file
146 	 * @param dataUrl
147 	 *            URL to <em>UAS data</em>
148 	 * @param versionUrl
149 	 *            URL to version information about the given <em>UAS data</em>
150 	 * @param charset
151 	 *            the character set in which the data should be read
152 	 * @param fallback
153 	 *            <em>UAS data</em> as fallback in case the data on the specified resource can not be read correctly
154 	 * @return new instance of {@link CachingXmlDataStore}
155 	 * @throws net.sf.qualitycheck.exception.IllegalNullArgumentException
156 	 *             if one of the given arguments is {@code null}
157 	 * @throws net.sf.qualitycheck.exception.IllegalNullArgumentException
158 	 *             if the given cache file can not be read
159 	 * @throws net.sf.qualitycheck.exception.IllegalStateOfArgumentException
160 	 *             if no URL can be resolved to the given given file
161 	 */
162 	@Nonnull
163 	public static CachingXmlDataStore createCachingXmlDataStore(@Nonnull final File cacheFile, @Nonnull final URL dataUrl,
164 			@Nonnull final URL versionUrl, @Nonnull final Charset charset, @Nonnull final DataStore fallback) {
165 		Check.notNull(cacheFile, "cacheFile");
166 		Check.notNull(charset, "charset");
167 		Check.notNull(dataUrl, "dataUrl");
168 		Check.notNull(fallback, "fallback");
169 		Check.notNull(versionUrl, "versionUrl");
170 
171 		final DataReader reader = new XmlDataReader();
172 		final DataStore fallbackDataStore = readCacheFileAsFallback(reader, cacheFile, charset, fallback);
173 		return new CachingXmlDataStore(reader, dataUrl, versionUrl, charset, cacheFile, fallbackDataStore);
174 	}

177 	 * Constructs a new instance of {@code CachingXmlDataStore} with the given arguments. The file used to cache the
178 	 * read in <em>UAS data</em> will be called from {@link CachingXmlDataStore#findOrCreateCacheFile()}. This file may
179 	 * be empty or filled with previously cached data in XML format. The file must be writable otherwise an exception
180 	 * will be thrown.
181 	 * 
182 	 * @param dataUrl
183 	 *            URL to <em>UAS data</em>
184 	 * @param versionUrl
185 	 *            URL to version information about the given <em>UAS data</em>
186 	 * @param charset
187 	 *            the character set in which the data should be read
188 	 * @param fallback
189 	 *            <em>UAS data</em> as fallback in case the data on the specified resource can not be read correctly
190 	 * @return new instance of {@link CachingXmlDataStore}
191 	 * @throws net.sf.qualitycheck.exception.IllegalNullArgumentException
192 	 *             if one of the given arguments is {@code null}
193 	 * @throws net.sf.qualitycheck.exception.IllegalStateOfArgumentException
194 	 *             if the given cache file can not be read
195 	 */
196 	@Nonnull
197 	public static CachingXmlDataStore createCachingXmlDataStore(@Nonnull final URL dataUrl, @Nonnull final URL versionUrl,
198 			@Nonnull final Charset charset, @Nonnull final DataStore fallback) {
199 		return createCachingXmlDataStore(findOrCreateCacheFile(), dataUrl, versionUrl, charset, fallback);
200 	}

202 	/**
203 	 * Removes the given cache file because it contains damaged content.
204 	 * 
205 	 * @param cacheFile
206 	 *            cache file to delete
207 	 */
208 	private static void deleteCacheFile(final File cacheFile) {
209 		try {
210 			if (cacheFile.delete()) {
211 				LOG.warn(String.format(MSG_CACHE_FILE_IS_DAMAGED_AND_DELETED, cacheFile.getPath()));
212 			} else {
213 				LOG.warn(String.format(MSG_CACHE_FILE_IS_DAMAGED, cacheFile.getPath()));
214 			}
215 		} catch (final Exception e) {
216 			LOG.warn(String.format(MSG_CACHE_FILE_IS_DAMAGED, cacheFile.getPath()));
217 		}
218 	}

220 	/**
221 	 * Gets the cache file for <em>UAS data</em> in the default temporary-file directory. If no cache file exists, a new
222 	 * empty file in the default temporary-file directory will be created, using the default prefix and suffix to
223 	 * generate its name.
224 	 * 
225 	 * @return file to cache read in <em>UAS data</em>
226 	 * @throws net.sf.qualitycheck.exception.IllegalStateOfArgumentException
227 	 *             if the cache file can not be created
228 	 */
229 	@Nonnull
230 	public static File findOrCreateCacheFile() {
231 		final File file = new File(CACHE_DIR, PREFIX + SUFFIX);
232 		if (!file.exists()) {
233 			try {
234 				file.createNewFile();
235 			} catch (final IOException e) {
236 				throw new IllegalStateOfArgumentException("Can not create a cache file.", e);
237 			}
238 		}
239 		return file;
240 	}

242 	/**
243 	 * Checks if the given file is empty.
244 	 * 
245 	 * @param file
246 	 *            the file that could be empty
247 	 * @return {@code true} when the file is accessible and empty otherwise {@code false}
248 	 * @throws net.sf.qualitycheck.exception.IllegalStateOfArgumentException
249 	 *             if an I/O error occurs
250 	 */
251 	private static boolean isEmpty(@Nonnull final File file, @Nonnull final Charset charset) {
252 		try {
253 			return FileUtil.isEmpty(file, charset);
254 		} catch (final IOException e) {
255 			throw new IllegalStateOfArgumentException("The given file could not be read.", e);
256 		}
257 	}

259 	/**
260 	 * Tries to read the content of specified cache file and returns them as fallback data store. If the cache file
261 	 * contains unexpected data the given fallback data store will be returned instead.
262 	 * 
263 	 * @param reader
264 	 *            data reader to read the given {@code dataUrl}
265 	 * @param cacheFile
266 	 *            file with cached <em>UAS data</em> in XML format or empty file
267 	 * @param versionUrl
268 	 *            URL to version information about the given <em>UAS data</em>
269 	 * @param charset
270 	 *            the character set in which the data should be read
271 	 * @param fallback
272 	 *            <em>UAS data</em> as fallback in case the data on the specified resource can not be read correctly
273 	 * @return a fallback data store
274 	 */
275 	private static DataStore readCacheFileAsFallback(@Nonnull final DataReader reader, @Nonnull final File cacheFile,
276 			@Nonnull final Charset charset, @Nonnull final DataStore fallback) {
277 		DataStore fallbackDataStore;
278 		if (!isEmpty(cacheFile, charset)) {
279 			final URL cacheFileUrl = UrlUtil.toUrl(cacheFile);
280 			try {
281 				fallbackDataStore = new CacheFileDataStore(reader.read(cacheFileUrl, charset), reader, cacheFileUrl, charset);
282 				LOG.debug(MSG_CACHE_FILE_IS_FILLED);
283 			} catch (final RuntimeException e) {
284 				fallbackDataStore = fallback;
285 				deleteCacheFile(cacheFile);
286 			}
287 		} else {
288 			fallbackDataStore = fallback;
289 			LOG.debug(MSG_CACHE_FILE_IS_EMPTY);
290 		}
291 		return fallbackDataStore;
292 	}

296 	 * 
297 	 * @param data

299 	 * @param reader
300 	 *            data reader to read the given {@code dataUrl}
301 	 * @param dataUrl
302 	 *            URL to <em>UAS data</em>
303 	 * @param versionUrl
304 	 *            URL to version information about the given <em>UAS data</em>
305 	 * @param charset
306 	 *            the character set in which the data should be read
307 	 * @param cacheFile
308 	 *            file with cached <em>UAS data</em> in XML format or an empty file
309 	 * @throws net.sf.qualitycheck.exception.IllegalNullArgumentException
310 	 *             if one of the given arguments is {@code null}
311 	 */
312 	private CachingXmlDataStore(@Nonnull final DataReader reader, @Nonnull final URL dataUrl, @Nonnull final URL versionUrl,
313 			@Nonnull final Charset charset, @Nonnull final File cacheFile, @Nonnull final DataStore fallback) {
314 		super(reader, dataUrl, versionUrl, charset, fallback);
315 		setUpdateOperation(new UpdateOperationWithCacheFileTask(this, cacheFile));
316 	}
