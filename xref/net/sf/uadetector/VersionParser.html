18  import java.util.ArrayList;
19  import java.util.Arrays;

28  /**
29   * This class is used to detect version information within <i>User-Agent</i> strings.
30   * 
31   * @author André Rouél
32   */
33  final class VersionParser {
34  
35  	/**
36  	 * Index number of the group in a matching {@link Pattern} which contains the extension/suffix of a version string
37  	 */
38  	private static final int EXTENSION_INDEX = 5;
39  
40  	/**
41  	 * Index number of the group in a matching {@link Pattern} which contains the first/major number of a version string
42  	 */
43  	private static final int MAJOR_INDEX = 1;
44  
45  	/**
46  	 * Regular expression to analyze a version number separated by a dot
47  	 */
48  	private static final Pattern VERSIONNUMBER = Pattern.compile("((\\d+)((\\.\\d+)+)?)");
49  
50  	/**
51  	 * Regular expression to analyze a version number separated by a dot with suffix
52  	 */
53  	private static final Pattern VERSIONNUMBER_WITH_SUFFIX = Pattern.compile(VERSIONNUMBER.pattern() + "((\\s|\\-|\\.|\\[|\\]|\\w+)+)?");
54  
55  	/**
56  	 * Regular expression to analyze segments of a version string, consisting of prefix, numeric groups and suffix
57  	 */
58  	private static final Pattern VERSIONSTRING = Pattern.compile("^" + VERSIONNUMBER_WITH_SUFFIX.pattern());
59  
60  	/**
61  	 * This method try to determine the version number of the operating system <i>Android</i> more accurately.
62  	 * 
63  	 * @param userAgent
64  	 *            user agent string
65  	 * @return more accurately identified version number or {@code null}
66  	 */
67  	static VersionNumber identifyAndroidVersion(@Nonnull final String userAgent) {
68  		VersionNumber version = VersionNumber.UNKNOWN;
69  		final List<Pattern> patterns = new ArrayList<Pattern>();
70  		patterns.add(Pattern.compile("Android\\s?((\\d+)((\\.\\d+)+)?(\\-(\\w|\\d)+)?);"));
71  		patterns.add(Pattern.compile("Android\\-((\\d+)((\\.\\d+)+)?(\\-(\\w|\\d)+)?);"));
72  		for (final Pattern pattern : patterns) {
73  			final Matcher m = pattern.matcher(userAgent);
74  			if (m.find()) {
75  				version = parseFirstVersionNumber(m.group(MAJOR_INDEX));
76  				break;
77  			}
78  		}
79  		return version;
80  	}

100 	 * This method try to determine the version number of an operating system of a <i>BSD</i> platform more accurately.
101 	 * 
102 	 * @param userAgent
103 	 *            user agent string
104 	 * @return more accurately identified version number or {@code null}
105 	 */
106 	static VersionNumber identifyBSDVersion(final String userAgent) {
107 		VersionNumber version = VersionNumber.UNKNOWN;
108 		final Pattern pattern = Pattern.compile("\\w+bsd\\s?((\\d+)((\\.\\d+)+)?((\\-|_)[\\w\\d\\-]+)?)", Pattern.CASE_INSENSITIVE);
109 		final Matcher m = pattern.matcher(userAgent);
110 		if (m.find()) {
111 			version = parseFirstVersionNumber(m.group(MAJOR_INDEX));
112 		}
113 		return version;
114 	}

117 	 * This method try to determine the version number of the operating system <i>iOS</i> more accurately.
118 	 * 
119 	 * @param userAgent
120 	 *            user agent string
121 	 * @return more accurately identified version number or {@code null}
122 	 */
123 	static VersionNumber identifyIOSVersion(final String userAgent) {
124 		VersionNumber version = VersionNumber.UNKNOWN;
125 		final List<Pattern> patterns = new ArrayList<Pattern>();
126 		patterns.add(Pattern.compile("iPhone OS\\s?((\\d+)((\\_\\d+)+)?) like Mac OS X"));
127 		patterns.add(Pattern.compile("CPU OS\\s?((\\d+)((\\_\\d+)+)?) like Mac OS X"));
128 		patterns.add(Pattern.compile("iPhone OS\\s?((\\d+)((\\.\\d+)+)?);"));
129 		for (final Pattern pattern : patterns) {
130 			final Matcher m = pattern.matcher(userAgent);
131 			if (m.find()) {
132 				version = parseFirstVersionNumber(m.group(MAJOR_INDEX).replaceAll("_", "."));
133 				break;
134 			}
135 		}
136 		return version;
137 	}

252 	/**
253 	 * Interprets a string with version information. The first occurrence of a version number in the string will be
254 	 * searched and processed.
255 	 * 
256 	 * @param text
257 	 *            string with version information
258 	 * @return an object of {@code VersionNumber}, never {@code null}
259 	 */
260 	static VersionNumber parseFirstVersionNumber(@Nonnull final String text) {
261 		Check.notNull(text, "text");
262 
263 		final Matcher matcher = VERSIONNUMBER_WITH_SUFFIX.matcher(text);
264 		String[] split = null;
265 		String ext = null;
266 		if (matcher.find()) {
267 			split = matcher.group(MAJOR_INDEX).split("\\.");
268 			ext = matcher.group(EXTENSION_INDEX);
269 		}
270 
271 		final String extension = ext == null ? VersionNumber.EMPTY_EXTENSION : trimRight(ext);
272 
273 		return split == null ? VersionNumber.UNKNOWN : new VersionNumber(Arrays.asList(split), extension);
274 	}
275 
276 	/**
277 	 * Interprets a string with version information. The last version number in the string will be searched and
278 	 * processed.
279 	 * 
280 	 * @param text
281 	 *            string with version information
282 	 * @return an object of {@code VersionNumber}, never {@code null}
283 	 */
284 	public static VersionNumber parseLastVersionNumber(@Nonnull final String text) {
285 		Check.notNull(text, "text");
286 
287 		final Matcher matcher = VERSIONNUMBER_WITH_SUFFIX.matcher(text);
288 		String[] split = null;
289 		String ext = null;
290 		while (matcher.find()) {
291 			split = matcher.group(MAJOR_INDEX).split("\\.");
292 			ext = matcher.group(EXTENSION_INDEX);
293 		}
294 
295 		final String extension = ext == null ? VersionNumber.EMPTY_EXTENSION : trimRight(ext);
296 
297 		return split == null ? VersionNumber.UNKNOWN : new VersionNumber(Arrays.asList(split), extension);
298 	}

300 	/**
301 	 * Try to determine the version number of the operating system by parsing the user agent string.
302 	 * 
303 	 * 
304 	 * @param family
305 	 *            family of the operating system
306 	 * @param userAgent
307 	 *            user agent string
308 	 * @return extracted version number
309 	 */
310 	public static VersionNumber parseOperatingSystemVersion(@Nonnull final OperatingSystemFamily family, @Nonnull final String userAgent) {
311 		Check.notNull(family, "family");
312 		Check.notNull(userAgent, "userAgent");
313 
314 		final VersionNumber v;
315 		if (OperatingSystemFamily.ANDROID == family) {
316 			v = identifyAndroidVersion(userAgent);
317 		} else if (OperatingSystemFamily.BADA == family) {
318 			v = identifyBadaVersion(userAgent);
319 		} else if (OperatingSystemFamily.BSD == family) {
320 			v = identifyBSDVersion(userAgent);
321 		} else if (OperatingSystemFamily.IOS == family) {
322 			v = identifyIOSVersion(userAgent);
323 		} else if (OperatingSystemFamily.JVM == family) {
324 			v = identifyJavaVersion(userAgent);
325 		} else if (OperatingSystemFamily.OS_X == family) {
326 			v = identifyOSXVersion(userAgent);
327 		} else if (OperatingSystemFamily.SYMBIAN == family) {
328 			v = identifySymbianVersion(userAgent);
329 		} else if (OperatingSystemFamily.WEBOS == family) {
330 			v = identifyWebOSVersion(userAgent);
331 		} else if (OperatingSystemFamily.WINDOWS == family) {
332 			v = identifyWindowsVersion(userAgent);
333 		} else {
334 			v = VersionNumber.UNKNOWN;
335 		}
336 		return v;
337 	}
338 
339 	/**
340 	 * Interprets a string with version information. The first found group will be taken and processed.
341 	 * 
342 	 * @param version
343 	 *            version as string
344 	 * @return an object of {@code VersionNumber}, never {@code null}
345 	 */
346 	public static VersionNumber parseVersion(@Nonnull final String version) {
347 		Check.notNull(version, "version");
348 
349 		VersionNumber result = new VersionNumber(new ArrayList<String>(0), version);
350 		final Matcher matcher = VERSIONSTRING.matcher(version);
351 		if (matcher.find()) {
352 			final List<String> groups = Arrays.asList(matcher.group(MAJOR_INDEX).split("\\."));
353 			final String extension = matcher.group(EXTENSION_INDEX) == null ? VersionNumber.EMPTY_EXTENSION : trimRight(matcher
354 					.group(EXTENSION_INDEX));
355 			result = new VersionNumber(groups, extension);
356 		}
357 
358 		return result;
359 	}

361 	/**
362 	 * Trims the whitespace at the end of the given string.
363 	 * 
364 	 * @param text
365 	 *            string to trim
366 	 * @return trimmed string
367 	 */
368 	private static String trimRight(@Nonnull final String text) {
369 		return text.replaceAll("\\s+$", "");
370 	}
