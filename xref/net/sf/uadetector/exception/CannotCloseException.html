16  package net.sf.uadetector.exception;
17  
18  /**
19   * Thrown to indicate that a {@link java.io.Closeable} cannot be closed.<br>
20   * <br>
21   * This exception is intended to tunnel the checked exception {@link java.io.IOException} during the call
22   * {@link java.io.Closeable#close()}.
23   * 
24   * @author André Rouél
25   */
26  public class CannotCloseException extends RuntimeException {
27  
28  	private static final long serialVersionUID = -8641033043995976022L;
29  
30  	protected static final String DEFAULT_MESSAGE = "Cannot close the given Closeable.";
31  
32  	protected static final String MESSAGE_WITH_INFO = "Cannot close the given Closeable: %s";
33  
34  	private static String format(final String url) {
35  		return String.format(MESSAGE_WITH_INFO, url);
36  	}
37  
38  	/**
39  	 * Constructs an {@code CannotCloseException} with the default message {@link CannotCloseException#DEFAULT_MESSAGE}.
40  	 */
41  	public CannotCloseException() {
42  		super(DEFAULT_MESSAGE);
43  	}
44  
45  	/**
46  	 * Constructs an {@code CannotCloseException} with the message {@link CannotCloseException#MESSAGE_WITH_INFO}
47  	 * including additional information.
48  	 * 
49  	 * @param info
50  	 *            additional information why a {@link java.io.Closeable} cannot be closed
51  	 */
52  	public CannotCloseException(final String info) {
53  		super(format(info));
54  	}
55  
56  	/**
57  	 * Constructs a new exception with the message {@link CannotCloseException#MESSAGE_WITH_INFO} including additional
58  	 * information.
59  	 * 
60  	 * @param info
61  	 *            additional information why a {@link java.io.Closeable} cannot be closed
62  	 * @param cause
63  	 *            the cause (which is saved for later retrieval by the {@link Throwable#getCause()} method). (A
64  	 *            {@code null} value is permitted, and indicates that the cause is nonexistent or unknown.)
65  	 */
66  	public CannotCloseException(final String info, final Throwable cause) {
67  		super(format(info), cause);
68  	}
69  
70  	/**
71  	 * Constructs a new exception with the default message {@link CannotCloseException#DEFAULT_MESSAGE}.
72  	 * 
73  	 * @param cause
74  	 *            the cause (which is saved for later retrieval by the {@link Throwable#getCause()} method). (A
75  	 *            {@code null} value is permitted, and indicates that the cause is nonexistent or unknown.)
76  	 */
77  	public CannotCloseException(final Throwable cause) {
78  		super(DEFAULT_MESSAGE, cause);
79  	}
80  
81  }
