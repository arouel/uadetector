16  package net.sf.uadetector.internal.data;
17  
18  import java.io.IOException;
19  
20  import net.sf.qualitycheck.exception.IllegalNullArgumentException;
21  
22  import static org.fest.assertions.Assertions.assertThat;
23  import org.junit.Test;
24  import org.xml.sax.InputSource;
25  import org.xml.sax.SAXException;
26  import org.xml.sax.SAXParseException;
27  
28  public class XmlDataHandlerTest {
29  
30  	@Test(expected = IllegalNullArgumentException.class)
31  	public void constructor_null() {
32  		new XmlDataHandler(null);
33  	}
34  
35  	/**
36  	 * This test reduces only some coverage noise.
37  	 */
38  	@Test
39  	public void logParsingIssue() {
40  		final String message = "msg";
41  		final String publicId = "public id";
42  		final String systemId = "system id";
43  		final int lineNumber = 123;
44  		final int columnNumber = 345;
45  		final IllegalArgumentException iae = new IllegalArgumentException();
46  		final SAXParseException e = new SAXParseException(message, publicId, systemId, lineNumber, columnNumber, iae);
47  		XmlDataHandler.logParsingIssue("Warning", e);
48  	}
49  
50  	@Test
51  	public void resolveEntity_knownUrl() throws IOException, SAXException {
52  		final XmlDataHandler handler = new XmlDataHandler(new DataBuilder());
53  		final InputSource input = handler.resolveEntity("publicId is irrelevant", XmlDataHandler.UASDATA_DEF_URL);
54  		assertThat(input).isNotNull();
55  	}
56  
57  	@Test(expected = SAXException.class)
58  	public void resolveEntity_unknownUrl() throws IOException, SAXException {
59  		final XmlDataHandler handler = new XmlDataHandler(new DataBuilder());
60  		handler.resolveEntity("publicId unknown", "systemId unknown");
61  	}
62  
63  }
