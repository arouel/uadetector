109 	public Data(@Nonnull final Set<Browser> browsers, @Nonnull final Map<Integer, SortedSet<BrowserPattern>> browserPatterns,
110 			@Nonnull final Map<Integer, BrowserType> browserTypes, @Nonnull final SortedMap<BrowserPattern, Browser> patternToBrowserMap,
111 			@Nonnull final Set<BrowserOperatingSystemMapping> browserToOperatingSystemMappings,
112 			@Nonnull final Set<OperatingSystem> operatingSystems,
113 			@Nonnull final Map<Integer, SortedSet<OperatingSystemPattern>> operatingSystemPatterns,
114 			@Nonnull final SortedMap<OperatingSystemPattern, OperatingSystem> patternToOperatingSystemMap,
115 			@Nonnull final List<Robot> robots, @Nonnull final Set<Device> devices,
116 			@Nonnull final Map<Integer, SortedSet<DevicePattern>> devicePatterns,
117 			@Nonnull final SortedMap<DevicePattern, Device> patternToDeviceMap, @Nonnull final String version) {
118 		Check.notNull(browsers, "browsers");
119 		Check.notNull(browserPatterns, "browserPatterns");
120 		Check.notNull(browserTypes, "browserTypes");
121 		Check.notNull(patternToBrowserMap, "patternToBrowserMap");
122 		Check.notNull(browserToOperatingSystemMappings, "browserToOperatingSystemMap");
123 		Check.notNull(operatingSystems, "operatingSystems");
124 		Check.notNull(operatingSystemPatterns, "operatingSystemPatterns");
125 		Check.notNull(patternToOperatingSystemMap, "patternToOperatingSystemMap");
126 		Check.notNull(robots, "robots");
127 		Check.notNull(devices, "devices");
128 		Check.notNull(devicePatterns, "devicePatterns");
129 		Check.notNull(patternToDeviceMap, "patternToDeviceMap");
130 		Check.notNull(version, "version");
131 
132 		this.browsers = Collections.unmodifiableSet(new HashSet<Browser>(browsers));
133 		this.browserPatterns = Collections.unmodifiableMap(new HashMap<Integer, SortedSet<BrowserPattern>>(browserPatterns));
134 		this.browserTypes = Collections.unmodifiableMap(new HashMap<Integer, BrowserType>(Check.notNull(browserTypes, "browserTypes")));
135 		this.patternToBrowserMap = Collections.unmodifiableSortedMap(new TreeMap<BrowserPattern, Browser>(patternToBrowserMap));
136 		this.browserToOperatingSystemMappings = Collections.unmodifiableSet(new HashSet<BrowserOperatingSystemMapping>(
137 				browserToOperatingSystemMappings));
138 		this.operatingSystems = Collections.unmodifiableSet(new HashSet<OperatingSystem>(operatingSystems));
139 		this.operatingSystemPatterns = Collections.unmodifiableMap(new HashMap<Integer, SortedSet<OperatingSystemPattern>>(
140 				operatingSystemPatterns));
141 		this.patternToOperatingSystemMap = Collections.unmodifiableSortedMap(new TreeMap<OperatingSystemPattern, OperatingSystem>(
142 				patternToOperatingSystemMap));
143 		this.robots = Collections.unmodifiableList(new ArrayList<Robot>(robots));
144 		this.devices = Collections.unmodifiableSet(new HashSet<Device>(devices));
145 		this.devicePatterns = Collections.unmodifiableMap(new HashMap<Integer, SortedSet<DevicePattern>>(devicePatterns));
146 		this.patternToDeviceMap = Collections.unmodifiableSortedMap(new TreeMap<DevicePattern, Device>(patternToDeviceMap));
147 		this.version = Check.notNull(version, "version");
148 	}

279 		result = prime * result + browserPatterns.hashCode();
280 		result = prime * result + browserTypes.hashCode();
281 		result = prime * result + patternToBrowserMap.hashCode();

294 	@Nonnull
295 	public String toStats() {
296 		final StringBuilder builder = new StringBuilder();
297 		builder.append("UAS data stats\n");
298 		builder.append("----------------------------------------------------------------");
299 		builder.append('\n');
300 		builder.append("version:\t\t");
301 		builder.append(version);
302 		builder.append('\n');
303 		builder.append("browser:\t\t");
304 		builder.append(browsers.size());
305 		builder.append('\n');
306 		final Map<String, AtomicInteger> browserByType = new HashMap<String, AtomicInteger>();
307 		for (final Browser browser : browsers) {
308 			final AtomicInteger counter = browserByType.get(browser.getType().getName());
309 			if (counter == null) {
310 				browserByType.put(browser.getType().getName(), new AtomicInteger(1));
311 			} else {
312 				counter.incrementAndGet();
313 			}
314 		}
315 		for (final Entry<String, AtomicInteger> entry : browserByType.entrySet()) {
316 			builder.append('\t');
317 			builder.append('\t');
318 			builder.append('\t');
319 			builder.append(entry.getKey());
320 			builder.append(":\t");
321 			builder.append(entry.getValue().get());
322 			builder.append('\n');
323 		}
324 		builder.append("browser patterns:\t");
325 		builder.append(patternToBrowserMap.size());
326 		builder.append('\n');
327 		builder.append("operating systems:\t");
328 		builder.append(operatingSystems.size());
329 		builder.append('\n');
330 		builder.append("os patterns:\t\t");
331 		builder.append(patternToOperatingSystemMap.size());
332 		builder.append('\n');
333 		builder.append("robots:\t\t\t");
334 		builder.append(robots.size());
335 		builder.append('\n');
336 		builder.append("devices:\t");
337 		builder.append(devices.size());
338 		builder.append('\n');
339 		builder.append("device patterns:\t");
340 		builder.append(patternToDeviceMap.size());
341 		builder.append('\n');
342 		builder.append("----------------------------------------------------------------");
343 		return builder.toString();
344 	}
