16  package net.sf.uadetector.datastore;
17  
18  import java.io.IOException;
19  import java.io.InputStream;
20  import java.io.InputStreamReader;
21  import java.io.LineNumberReader;
22  import java.net.URL;
23  import java.nio.charset.Charset;
24  import java.util.concurrent.ExecutorService;
25  import java.util.regex.Pattern;
26  
27  import javax.annotation.Nonnull;
28  import javax.annotation.Nullable;
29  
30  import net.sf.qualitycheck.Check;
31  import net.sf.uadetector.internal.util.ExecutorServices;
32  
33  import org.slf4j.Logger;
34  import org.slf4j.LoggerFactory;
35  
36  /**
37   * Provides a basic implementation to update UAS data in the background when calling {@link #run()}.
38   * 
39   * @author André Rouél
40   */
41  public abstract class AbstractUpdateOperation implements UpdateOperation {
42  
43  	/**
44  	 * Defines an empty version string
45  	 */
46  	private static final String EMPTY_VERSION = "";
47  
48  	/**
49  	 * Corresponding default logger for this class
50  	 */
51  	private static final Logger LOG = LoggerFactory.getLogger(AbstractUpdateOperation.class);
52  
53  	/**
54  	 * Message for the log when no update is available.<br>
55  	 * <br>
56  	 * <b>Message sample</b>: No update available. Current version is '<em>20120301-01</em>'.<br>
57  	 * <b>First placeholder</b>: current version
58  	 */
59  	private static final String MSG_NO_UPDATE_AVAILABLE = "No update available. Current version is '%s'.";
60  
61  	/**
62  	 * Message for the log when an online update check is not possible.<br>
63  	 * <br>
64  	 * <b>Message sample</b>: Can not check for an updated version. Are you sure you have an established internet
65  	 * connection?
66  	 */
67  	private static final String MSG_NO_UPDATE_CHECK_POSSIBLE = "Can not check for an updated version. Are you sure you have an established internet connection?";
68  
69  	/**
70  	 * Message for the log when an exception occur during the update check.<br>
71  	 * <br>
72  	 * <b>Message sample</b>: Can not check for an updated version: <em>java.net.ConnectException</em>:
73  	 * <em>Connection refused</em><br>
74  	 * <b>First placeholder</b>: class name of exception<br>
75  	 * <b>Second placeholder</b>: exception message
76  	 */
77  	private static final String MSG_NO_UPDATE_CHECK_POSSIBLE__DEBUG = "Can not check for an updated version: %s: %s";
78  
79  	/**
80  	 * Message for the log when an update is available.<br>
81  	 * <br>
82  	 * <b>Message sample</b>: An update is available. Current version is '<em>20120301-01</em>' and remote version is '
83  	 * <em>20120401-01</em>'.<br>
84  	 * <b>First placeholder</b>: current version<br>
85  	 * <b>Second placeholder</b>: new remote version
86  	 */
87  	private static final String MSG_UPDATE_AVAILABLE = "An update is available. Current version is '%s' and remote version is '%s'.";
88  
89  	/**
90  	 * Pattern of a typical version of <i>UAS data</i>
91  	 */
92  	private static final Pattern VERSION_PATTERN = Pattern.compile("\\d{8}\\-\\d{2}");
93  
94  	/**
95  	 * Checks a given newer version against an older one.
96  	 * 
97  	 * @param newer
98  	 *            possible newer version
99  	 * @param older
100 	 *            possible older version
101 	 * @return {@code true} if the first argument is newer than the second argument, otherwise {@code false}
102 	 */
103 	static boolean hasUpdate(final String newer, final String older) {
104 		return VERSION_PATTERN.matcher(newer).matches() && VERSION_PATTERN.matcher(older).matches() ? newer.compareTo(older) > 0 : false;
105 	}
106 
107 	/**
108 	 * Reads the current User-Agent data version from <a
109 	 * href="http://user-agent-string.info">http://user-agent-string.info</a>.
110 	 * 
111 	 * @param url
112 	 *            a URL which the version information can be loaded
113 	 * @return a version string or {@code null}
114 	 * @throws IOException
115 	 *             if an I/O exception occurs
116 	 */
117 	@Nullable
118 	private static String retrieveRemoteVersion(@Nonnull final URL url, @Nonnull final Charset charset) throws IOException {
119 		final InputStream stream = url.openStream();
120 		final InputStreamReader reader = new InputStreamReader(stream, charset);
121 		final LineNumberReader lnr = new LineNumberReader(reader);
122 		final String line = lnr.readLine();
123 		lnr.close();
124 		reader.close();
125 		stream.close();
126 		return line;
127 	}
128 
129 	/**
130 	 * {@link ExecutorService} to run the update operation of the UAS data in background
131 	 */
132 	private final ExecutorService executorService = ExecutorServices.createBackgroundExecutor();
133 
134 	/**
135 	 * Time of last update check in milliseconds
136 	 */
137 	private long lastUpdateCheck = 0;
138 
139 	/**
140 	 * The data store for instances that implements {@link net.sf.uadetector.internal.data.Data}
141 	 */
142 	private final RefreshableDataStore store;
143 
144 	public AbstractUpdateOperation(@Nonnull final RefreshableDataStore dataStore) {
145 		Check.notNull(dataStore, "dataStore");
146 		store = dataStore;
147 	}
148 
149 	/**
150 	 * Shortcut to get the current version of the UAS data in the {@link net.sf.uadetector.datastore.DataStore}.
151 	 * 
152 	 * @return current version of UAS data
153 	 */
154 	@Nonnull
155 	private String getCurrentVersion() {
156 		return store.getData().getVersion();
157 	}
158 
159 	/**
160 	 * Gets the time of the last update check in milliseconds.
161 	 * 
162 	 * @return time of the last update check in milliseconds
163 	 */
164 	@Override
165 	public long getLastUpdateCheck() {
166 		return lastUpdateCheck;
167 	}
168 
169 	/**
170 	 * Fetches the current version information over HTTP and compares it with the last version of the most recently
171 	 * imported data.
172 	 * 
173 	 * @return {@code true} if an update exists, otherwise {@code false}
174 	 */
175 	protected boolean isUpdateAvailable() {
176 		boolean result = false;
177 		String version = EMPTY_VERSION;
178 		try {
179 			version = retrieveRemoteVersion(store.getVersionUrl(), store.getCharset());
180 		} catch (final IOException e) {
181 			LOG.info(MSG_NO_UPDATE_CHECK_POSSIBLE);
182 			LOG.debug(String.format(MSG_NO_UPDATE_CHECK_POSSIBLE__DEBUG, e.getClass().getName(), e.getLocalizedMessage()));
183 		}
184 		if (hasUpdate(version, getCurrentVersion())) {
185 			LOG.debug(String.format(MSG_UPDATE_AVAILABLE, getCurrentVersion(), version));
186 			result = true;
187 		} else {
188 			LOG.debug(String.format(MSG_NO_UPDATE_AVAILABLE, getCurrentVersion()));
189 		}
190 		lastUpdateCheck = System.currentTimeMillis();
191 		return result;
192 	}
193 
194 	/**
195 	 * Executes the update at some time in the future (as soon as possible) within a new thread.
196 	 */
197 	@Override
198 	public void run() {
199 		executorService.execute(new Runnable() {
200 			@Override
201 			public void run() {
202 				call();
203 			}
204 		});
205 	}
206 
207 	/**
208 	 * Shuts down the corresponding background executor as soon as possible, but at the latest specified default time.
209 	 * 
210 	 * @see ExecutorServices#shutdown(ExecutorService)
211 	 */
212 	@Override
213 	public void shutdown() {
214 		ExecutorServices.shutdown(executorService);
215 	}
216 
217 }
