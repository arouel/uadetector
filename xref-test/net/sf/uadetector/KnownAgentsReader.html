16  package net.sf.uadetector;
17  
18  import java.io.File;
19  import java.io.FileNotFoundException;
20  import java.io.FileOutputStream;
21  import java.io.IOException;
22  import java.io.InputStream;
23  import java.io.InputStreamReader;
24  import java.io.LineNumberReader;
25  import java.io.OutputStreamWriter;
26  import java.util.HashMap;
27  import java.util.Map;
28  
29  import net.sf.uadetector.service.UADetectorServiceFactory;
30  
31  import org.slf4j.Logger;
32  import org.slf4j.LoggerFactory;
33  
34  public class KnownAgentsReader {
35  
36  	/**
37  	 * Character set of the file
38  	 */
39  	private static final String CHARSET = "UTF-8";
40  
41  	/**
42  	 * File to read
43  	 */
44  	private static final String FILE = "known_agents.txt";
45  
46  	/**
47  	 * Default log
48  	 */
49  	private static final Logger LOG = LoggerFactory.getLogger(KnownAgentsReader.class);
50  
51  	/**
52  	 * Parser to detect informations about an user agent
53  	 */
54  	private static final UserAgentStringParser parser = UADetectorServiceFactory.getResourceModuleParser();
55  
56  	/**
57  	 * CSV printer for detected version numbers to be particularly examined
58  	 */
59  	private final OutputStreamWriter printerForUncertainVersions;
60  
61  	private final Map<OperatingSystemFamily, OutputStreamWriter> printers = new HashMap<OperatingSystemFamily, OutputStreamWriter>();
62  
63  	private static final char DELIMITER = ',';
64  
65  	private static final char ESCAPE = '"';
66  
67  	private static final char NEWLINE = '\n';
68  
69  	/**
70  	 * Constructs an instance
71  	 * 
72  	 * @throws FileNotFoundException
73  	 */
74  	public KnownAgentsReader() throws FileNotFoundException {
75  		final File directory = new File("target/test-os-version-detection");
76  		final String extension = ".csv";
77  		if (!directory.exists()) {
78  			directory.mkdir();
79  		}
80  		for (final OperatingSystemFamily family : OperatingSystemFamily.values()) {
81  			printers.put(family, new OutputStreamWriter(new FileOutputStream(new File(directory.getPath() + "/" + family.toString()
82  					+ extension))));
83  		}
84  		printerForUncertainVersions = new OutputStreamWriter(new FileOutputStream(new File(directory.getPath() + "/" + "_uncertain"
85  				+ extension)));
86  	}
87  
88  	/**
89  	 * Prints the given informations a CSV printer.
90  	 * 
91  	 * @param userAgentString
92  	 *            user agent string
93  	 * @param family
94  	 *            operating system family
95  	 * @param name
96  	 *            operating system name
97  	 * @param version
98  	 *            detected version of the operating system
99  	 * @param printer
100 	 *            CSV printer to print on
101 	 * @throws IOException
102 	 */
103 	private void print(final String userAgentString, final OperatingSystemFamily family, final String name, final String version,
104 			final OutputStreamWriter printer) throws IOException {
105 		printer.append(ESCAPE);
106 		printer.append(family.toString());
107 		printer.append(ESCAPE);
108 		printer.append(DELIMITER);
109 		printer.append(ESCAPE);
110 		printer.append(name);
111 		printer.append(ESCAPE);
112 		printer.append(DELIMITER);
113 		printer.append(ESCAPE);
114 		printer.append(version);
115 		printer.append(ESCAPE);
116 		printer.append(DELIMITER);
117 		printer.append(ESCAPE);
118 		printer.append(userAgentString);
119 		printer.append(ESCAPE);
120 		printer.append(NEWLINE);
121 		printer.flush();
122 	}
123 
124 	public void read() {
125 		final InputStream stream = UserAgentStringParserIntegrationTest.class.getClassLoader().getResourceAsStream(FILE);
126 		try {
127 			final LineNumberReader reader = new LineNumberReader(new InputStreamReader(stream, CHARSET));
128 			String userAgentString = null;
129 			do {
130 				userAgentString = reader.readLine();
131 				if (userAgentString != null) {
132 					final ReadableUserAgent agent = parser.parse(userAgentString);
133 					final OperatingSystemFamily family = agent.getOperatingSystem().getFamily();
134 					final String name = agent.getOperatingSystem().getName();
135 					final String version = agent.getOperatingSystem().getVersionNumber().toVersionString();
136 					final OutputStreamWriter printer = printers.get(family);
137 					if (OperatingSystemFamily.WINDOWS == family) {
138 						if (name.equals("Windows 3.x") && version.equals("3.1")) {
139 							print(userAgentString, family, name, version, printer);
140 							continue;
141 						} else if (name.equals("Windows 95") && version.equals("95")) {
142 							print(userAgentString, family, name, version, printer);
143 							continue;
144 						} else if (name.equals("Windows 98") && version.equals("98")) {
145 							print(userAgentString, family, name, version, printer);
146 							continue;
147 						} else if (name.equals("Windows NT") && version.equals("4.0")) {
148 							print(userAgentString, family, name, version, printer);
149 							continue;
150 						} else if (name.equals("Windows NT") && version.equals("4.1")) {
151 							print(userAgentString, family, name, version, printer);
152 							continue;
153 						} else if (name.equals("Windows ME") && version.equals("4.90")) {
154 							print(userAgentString, family, name, version, printer);
155 							continue;
156 						} else if (name.equals("Windows 2000") && version.equals("5.0")) {
157 							print(userAgentString, family, name, version, printer);
158 							continue;
159 						} else if (name.equals("Windows XP") && version.equals("5.1")) {
160 							print(userAgentString, family, name, version, printer);
161 							continue;
162 						} else if (name.equals("Windows 2003 Server") && version.equals("5.2")) {
163 							print(userAgentString, family, name, version, printer);
164 							continue;
165 						} else if (name.equals("Windows Vista") && version.equals("6.0")) {
166 							print(userAgentString, family, name, version, printer);
167 							continue;
168 						} else if (name.equals("Windows 7") && version.equals("6.1")) {
169 							print(userAgentString, family, name, version, printer);
170 							continue;
171 						} else if (name.equals("Windows Phone 7") && version.matches("7((\\.\\d+)+)?")) {
172 							print(userAgentString, family, name, version, printer);
173 							continue;
174 						} else if (name.equals("Windows 8") && version.equals("6.2")) {
175 							print(userAgentString, family, name, version, printer);
176 							continue;
177 						} else if (name.equals("Windows") && version.matches("(\\d+)((\\.\\d+)+)?")) {
178 							print(userAgentString, family, name, version, printer);
179 							continue;
180 						} else if (VersionNumber.UNKNOWN.equals(agent.getOperatingSystem().getVersionNumber())) {
181 							print(userAgentString, family, name, version, printer);
182 							continue;
183 						} else {
184 							print(userAgentString, family, name, version, printerForUncertainVersions);
185 						}
186 					} else {
187 						print(userAgentString, family, name, version, printer);
188 					}
189 				}
190 			} while (userAgentString != null);
191 
192 		} catch (final IOException e) {
193 			LOG.warn(e.getLocalizedMessage(), e);
194 		}
195 	}
196 }
